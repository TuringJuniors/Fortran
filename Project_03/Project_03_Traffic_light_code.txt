class LED
{
    public:
    int pin;
    void on();
    void off();
};

class Button
{
    public:
    int pin;
    int digitalRead(pin);
};

LED green_car,red_car,yellow_car,green_pedestrian,wait_pedestrian,red_pedestrian;
Button button_pedestrian,button_bus;

void LED::on()
{
    digitalWrite(pin,HIGH);
}

void LED::off()
{
    digitalWrite(pin,LOW);
}

void UPDATE_INPUT()
{

}
